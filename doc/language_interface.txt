/language/source_list  数据源列表
params:
{
}
return:
{
   "status":0,
   "msg":"success",
   "data": {
       "sources": ["chatgpt","百川","viccuna"]
   }
}

/language/refresh 重新开始会话
params:
{
    "source": "百川"
}
return:
{
    "status":0,
    "msg":"success"
}

/language/send_msg 发送消息
params:
{
    "source": "百川",
    "msg": "你好啊"
}
return:
{
    "status":0,
    "msg":"success"
}

/language/receive_msg 接收消息
params:
{
    "source": "百川"
}
return:
{
    "status":0,
    "msg":"success",
    "data": {
        "flag": true,
        "recv_msg": "有什么可以帮到你的吗?"
    }
}

/language/fetch_task 获取任务
params:
{
    "source": "百川"
}
return:
{
    "status":0,
    "msg":"success",
    "data": {
        "conversation": "[]",
        "task_id": "1111"
    }
}

/language/reply_task 返回任务结果
params:
{
    "source": "百川",
    "recv_msg": "你好啊",
    "task_id": "1111"
}
return:
{
    "status":0,
    "msg":"success"
}
